FROM sales
GROUP BY cust; 1, 2, 3
SUCH THAT 1.state = 'NY' AND 1.cust = cust,
          2.state = 'NJ' AND 2.cust = cust,
          3.state = 'CT' AND 3.cust = cust
HAVING sum(1.quant) > 2 * sum(2.quant) OR avg(1.quant) > avg(3.quant)
